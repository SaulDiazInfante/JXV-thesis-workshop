                    \\

                \item
                    $
                        \displaystyle
                        \lim_{t \to \infty}
                        P_{I_0,t}(\cdot)
                        = \mathbb{P}_{\infty}(\cdot)
                    $ en distribución.
            \end{list}
        \end{greenbox}
